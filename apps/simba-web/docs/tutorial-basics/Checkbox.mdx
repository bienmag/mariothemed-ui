import { Checkbox } from "tailwind-simba-ui";
import styles from "./pages.module.css";

Checkbox component is used in forms when a user needs to select multiple values from several options.

### **Import**

```js
import { Checkbox } from "tailwind-simba-ui";
```

### **Usage**

export const CheckboxUsage = ({ defaultChecked }) => (
  <div className="flex h-16 items-center border-2	rounded-xl mb-4">
    <div className="pr-2 pl-2">
      <Checkbox color="yellow" defaultChecked></Checkbox>
    </div>
  </div>
);

<CheckboxUsage></CheckboxUsage>

```js
<Checkbox defaultChecked></Checkbox>
```

### **Disabled Checkbox**

export const DisabledCheckbox = () => (

<DisabledCheckbox></DisabledCheckbox>

```js
<Checkbox disabled color="yellow"></Checkbox>

```

### **Checkbox with custom color**

You can override the color scheme of the Checkbox to other color key: yellow, red, blue or green.

export const CheckboxColors = ({ color }) => (
  <div className="flex h-24 items-center border-2	rounded-xl mb-4">
    <div className="pr-2 pl-2">
      <Checkbox color="yellow"></Checkbox>
    </div>
    <div className="pr-2">
      <Checkbox color="red"></Checkbox>
    </div>
    <div className="pr-2">
      <Checkbox color="blue"></Checkbox>
    </div>
    <div>
      <Checkbox color="green"></Checkbox>
    </div>
  </div>
);

<CheckboxColors></CheckboxColors>

```js
<div>
  <MyButton color="yellow"></MyButton>
  <MyButton color="red"></MyButton>
  <MyButton color="blue"></MyButton>
  <MyButton color="green"></MyButton>
</div>
```

### **Checkbox Sizes**

export const CheckboxSizes = ({ size }) => (
  <div className="flex h-24 items-center border-2	rounded-xl mb-4">
    <div className="pr-2 pl-2">
      <Checkbox color="yellow" size="xs"></Checkbox>
    </div>
    <div className="pr-2">
      <Checkbox color="yellow" size="md"></Checkbox>
    </div>
    <div>
      <Checkbox color="yellow" size="lg"></Checkbox>
    </div>
  </div>
);

Use the size prop to change the size of the button. You can set the value to **xs**, **md**, or **lg**.

<CheckboxSizes></CheckboxSizes>

```js
<div>
  <Checkbox color="yellow" size="xs"></Checkbox>
  <Checkbox color="yellow" size="sm"></Checkbox>
  <Checkbox color="yellow" size="md"></Checkbox>
  <Checkbox color="yellow" size="lg"></Checkbox>
</div>
```
